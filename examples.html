example 1
Basic example of a round loading animation 
var round = new SavageLoader($('#area1'));
example 2
Altering the timing of the animation; moveToNextDelay = the time it takes for the animation to move between images, fadeDelay = the time it takes for each image to disappear. 
var round = new SavageLoader($('#area2'), {
    moveToNextDelay: 100,
    fadeDelay: 600
});
example 3
Using a different type of fade ('hide' vs 'fade'). 
var round = new SavageLoader($('#area3'), {
    moveToNextDelay: 100,
    fadeDelay: 700,
    fadeType: "hide"
});
example 4
A horizontal loading animation. 
var round = new SavageLoader($('#area4'), {
    fadeDelay: 1300,
    loaderType: "horizontal"
});
example 5
A smaller version of the above. 
var round = new SavageLoader($('#area5'), {
    fadeDelay: 1000,
    loaderType: "horizontal",
    imageCount: 5,
    turnDelay: 100
});
